import { Component } from "react";

class FirstComp extends Component{
    state = {
        stateVersion : 10
    }
    version = 0;
    render(){
        return <div>
                  <h2>{ this.props.title }</h2>
                  <h2>State Version : { this.state.stateVersion }</h2>
                  <h2>Version : { this.version }</h2>
                  <button onClick={ ()=> { 
                    // this.version = this.version+1; 
                    // // this.forceUpdate();
                    // console.log(this.version);
                    this.setState({ stateVersion : this.state.stateVersion+1 })
                } 
                    }>Click to change the version</button>
               </div>
    }
}

export default FirstComp;
